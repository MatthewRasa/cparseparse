144 template<class T, class Convert_Func>
145 T parse_numeric_arg(const std::string &value, Convert_Func &&convert_func) const {
146 try {
147 const auto n_value = convert_func(value);
148 if (n_value < std::numeric_limits<T>::lowest() || n_value > std::numeric_limits<T>::max())
149 throw std::out_of_range{""};
150 return n_value;
151 } catch (const std::invalid_argument &ex) {
152 throw std::runtime_error{errstr("'", m_name, "' must be of integral type")};
153 } catch (const std::out_of_range &ex) {
154 throw std::runtime_error{errstr("'", m_name, "' must be in range [", std::numeric_limits<T>::min(), ",", std::numeric_limits<T>::max(), "]")};
155 }
156 }
