108 Positional_Info &add_positional(std::string name) {
109 if (!valid_positional_name(name))
110 throw std::logic_error{lerrstr("invalid positional argument name '", name, "'")};
111 if (m_optional_args.find(name) != m_optional_args.end())
112 throw std::logic_error{lerrstr("positional argument name conflicts with optional argument reference name '", name, "'")};
113 const auto pair = m_positional_args.emplace(name, make_unique<Positional_Info>(name));

116 m_positional_order.push_back(std::move(name));

168 Optional_Info &add_optional(std::string flag, String &&long_name, Optional_Info::Type type = Optional_Info::Type::SINGLE) {
169 const auto formatted_name = format_flag_name(flag);
170 if (!formatted_name)
171 throw std::logic_error{lerrstr("invalid flag name '", flag, "'")};
172 if (m_flags.find(formatted_name) != m_flags.end())
173 throw std::logic_error{lerrstr("duplicate flag name '", flag, "'")};
174 auto &optional = add_optional(std::forward<String>(long_name), std::move(type));

499 Matched_Opts match_opt_args(const std::vector<Opt_Token> &arg_tokens) const {
500 auto optional_args = init_matched_opt_args(arg_tokens);
501 const std::string *p_last_optional;
502 for (const auto &token : arg_tokens) {
503 switch (token.type) {
504 case Opt_Token::Type::OPT_NAME:
505 p_last_optional = &token.value;
506 break;
507 case Opt_Token::Type::OPT_VAL:
508 optional_args[*p_last_optional].push_back(std::move(token.value));
509 break;
510 case Opt_Token::Type::FLAG:
511 optional_args[token.value].push_back("true");
512 break;
513 }
514 }
515 return optional_args;
516 }

521 Matched_Opts init_matched_opt_args(const std::vector<Opt_Token> &arg_tokens) const {
522 Matched_Opts optional_args;
523 const auto optional_counts = count_opt_args(arg_tokens);
524 optional_args.reserve(optional_counts.size());
525 for (const auto &pair : optional_counts)
526 optional_args[pair.first].reserve(pair.second);
527 return optional_args;
528 }

546 std::pair<std::vector<const char *>, std::vector<Opt_Token>> tokenize_args(int argc, const char **argv) const {
547 const std::vector<const char *> user_args(argv + 1, argv + argc);
548 std::vector<const char *> pos_args;
549 pos_args.reserve(user_args.size());
550 std::vector<Opt_Token> opt_tokens;
551 opt_tokens.reserve(user_args.size());
552
553 std::unordered_set<std::string> optional_names;
554 for (auto it = user_args.begin(); it != user_args.end(); ++it) {
555 auto optional_name = lookup_formatted_option_name(*it);
556 if (optional_name.empty()) {
557 pos_args.push_back(*it);
558 } else {
559 if (prematch_optional_arg(optional_name, optional_names.find(optional_name) != optional_names.end(), *std::next(it), *(user_args.end()))) {
560 opt_tokens.push_back(Opt_Token{Opt_Token::Type::OPT_NAME, optional_name});
561 opt_tokens.push_back(Opt_Token{Opt_Token::Type::OPT_VAL, *(++it)});
562 } else {
563 opt_tokens.push_back(Opt_Token{Opt_Token::Type::FLAG, optional_name});
564 }
565 optional_names.insert(optional_name);
566 }
567 }
568 if (pos_args.size() < m_positional_order.size())
569 throw std::runtime_error{errstr("requires positional argument '", m_positional_order[pos_args.size()], "'")};
570
571 return std::make_pair(pos_args, opt_tokens);
572 }

578 std::string lookup_formatted_option_name(const std::string &option_name) const {
579 const auto flag_name = format_flag_name(option_name);
580 if (flag_name) {
581 const auto flag_it = m_flags.find(flag_name);
582 if (flag_it == m_flags.end())
583 throw std::runtime_error{errstr("invalid flag '", option_name, "', pass --help to display possible options")};
584 return flag_it->second;
585 }
586 return format_option_name(option_name);
587 }

592 bool prematch_optional_arg(const std::string &option_name, bool repeated, const char *next_arg, const char *args_end) const {
593 if (m_auto_help && option_name == "help")
594 m_help_handler(*this);
595
596 auto it = m_optional_args.find(option_name);
597 if (it == m_optional_args.end())
598 throw std::runtime_error{errstr("invalid option '", option_name, "', pass --help to display possible options")};
599
600 if (it->second->type() == Optional_Info::Type::FLAG) {
601 if (repeated)
602 throw std::runtime_error{errstr("'", option_name, "' should only be specified once")};
603 return false;
604 } else {
605 if (next_arg == args_end)
606 throw std::runtime_error{errstr("'", option_name, "' requires a value")};
607 if (valid_option_name(next_arg))
608 throw std::runtime_error{errstr("'", option_name, "' requires a value")};
609 if (it->second->type() != Optional_Info::Type::APPEND && repeated)
610 throw std::runtime_error{errstr("'", option_name, "' should only be specified once")};
611 return true;
612 }
613 }
