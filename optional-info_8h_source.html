209 template<class T, bool has_default>
210 T as_type_at(std::size_t idx, T &&default_val) const {
211 if (exists())
212 return parse_as_type<T>(value(idx));
213 IF_CONSTEXPR (has_default)
214 return std::forward<T>(default_val);
215 if (m_type == Type::FLAG)
